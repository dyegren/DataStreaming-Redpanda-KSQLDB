-- paso 1
-- se crea el stream o tabla para consultar
CREATE STREAM STOCK_UPDATES (
    SYMBOL STRING,
    PRICE DOUBLE,
    VOLUME INTEGER,
    TIMESTAMP BIGINT
) WITH (
    KAFKA_TOPIC='stock_updates',
    PARTITIONS=1,      
    VALUE_FORMAT='JSON'
);

CREATE STREAM bolsa (
	symbol VARCHAR, 
	price DOUBLE, 
	volume DOUBLE, 
	timestamp VARCHAR
) 
WITH(
kafka_topic='stock-updates', 
value_format='json', 
partitions=1);

-- paso 2
se crea la vista para consultas
CREATE TABLE avg_price AS
SELECT symbol, 
SUM(price * volume) / SUM(volume) AS weighted_avg_price 
FROM bolsa GROUP BY symbol EMIT CHANGES;

CREATE TABLE num_transactions AS
SELECT symbol, 
COUNT(*) AS num_transactions 
FROM bolsa
GROUP BY symbol 
EMIT CHANGES;

CREATE TABLE max_price AS
SELECT symbol, 
MAX(price) AS max_price 
FROM bolsa 
GROUP BY symbol 
EMIT CHANGES;

CREATE TABLE min_price AS
SELECT symbol, 
MIN(price) AS min_price 
FROM bolsa 
GROUP BY symbol 
EMIT CHANGES;

--verificacion
CREATE TABLE prueba AS
SELECT symbol, 
price,
volume
FROM bolsa
EMIT CHANGES;

SELECT symbol,
count(*) AS total_transacciones
FROM bolsa
